def nodeModulesPath = new File(projectDir, "..")

    nodeModulesPath = new File(projectDir, "../../node_modules")
}

def autolinkingPath = new File(nodeModulesPath, "node_modules/expo-modules-autolinking/scripts/autolinking.gradle")

}
